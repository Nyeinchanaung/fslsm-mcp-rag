### Task 0.4 — Unified LLM Client

**`src/utils/llm_client.py`** — Single interface for all three LLM providers:

```python
from enum import Enum
from dataclasses import dataclass
from typing import Optional
import openai
import anthropic

class LLMProvider(str, Enum):
    OPENAI = "openai"
    ANTHROPIC = "anthropic"
    OLLAMA = "ollama"           # local Llama 3 via Ollama

@dataclass
class LLMResponse:
    content: str
    prompt_tokens: int
    completion_tokens: int
    model: str

class LLMClient:
    def __init__(self, model: str, provider: LLMProvider, temperature: float = 0.3):
        self.model = model
        self.provider = provider
        self.temperature = temperature
        if provider == LLMProvider.OPENAI:
            from config.settings import settings
            self._client = openai.OpenAI(api_key=settings.openai_api_key)
        elif provider == LLMProvider.ANTHROPIC:
            from config.settings import settings
            self._client = anthropic.Anthropic(api_key=settings.anthropic_api_key)

    def chat(self, system: str, user: str, max_tokens: int = 512) -> LLMResponse:
        if self.provider == LLMProvider.OPENAI:
            return self._openai_chat(system, user, max_tokens)
        elif self.provider == LLMProvider.ANTHROPIC:
            return self._anthropic_chat(system, user, max_tokens)
        elif self.provider == LLMProvider.OLLAMA:
            return self._ollama_chat(system, user, max_tokens)

    def _openai_chat(self, system, user, max_tokens) -> LLMResponse:
        resp = self._client.chat.completions.create(
            model=self.model,
            messages=[{"role": "system", "content": system},
                      {"role": "user", "content": user}],
            temperature=self.temperature,
            max_tokens=max_tokens,
        )
        return LLMResponse(
            content=resp.choices[0].message.content,
            prompt_tokens=resp.usage.prompt_tokens,
            completion_tokens=resp.usage.completion_tokens,
            model=self.model,
        )

    def _anthropic_chat(self, system, user, max_tokens) -> LLMResponse:
        resp = self._client.messages.create(
            model=self.model,
            max_tokens=max_tokens,
            temperature=self.temperature,
            system=system,
            messages=[{"role": "user", "content": user}],
        )
        return LLMResponse(
            content=resp.content[0].text,
            prompt_tokens=resp.usage.input_tokens,
            completion_tokens=resp.usage.output_tokens,
            model=self.model,
        )

    def _ollama_chat(self, system, user, max_tokens) -> LLMResponse:
        import urllib.request, json
        payload = json.dumps({
            "model": self.model,
            "messages": [{"role": "system", "content": system},
                         {"role": "user", "content": user}],
            "stream": False,
            "options": {"temperature": self.temperature, "num_predict": max_tokens},
        }).encode()
        req = urllib.request.Request(
            "http://localhost:11434/api/chat",
            data=payload, headers={"Content-Type": "application/json"}, method="POST"
        )
        with urllib.request.urlopen(req) as r:
            data = json.loads(r.read())
        return LLMResponse(
            content=data["message"]["content"],
            prompt_tokens=data.get("prompt_eval_count", 0),
            completion_tokens=data.get("eval_count", 0),
            model=self.model,
        )
```

### Task 1.2 — Embed Model Setup

**`src/utils/embedding_client.py`**:

```python
from sentence_transformers import SentenceTransformer
import numpy as np
from config.settings import settings

class EmbeddingClient:
    def __init__(self, model_name: str = None):
        self.model_name = model_name or settings.embed_model
        self._model = SentenceTransformer(self.model_name)

    def embed(self, texts: list[str]) -> np.ndarray:
        return self._model.encode(texts, batch_size=64,
                                  show_progress_bar=True, normalize_embeddings=True)

    def embed_one(self, text: str) -> np.ndarray:
        return self._model.encode([text], normalize_embeddings=True)[0]
```

### Task 1.3 — Ingest Corpus Chunks → DB + FAISS

**`scripts/ingest_d2l.py`** — reads `data/raw/d2l/d2l_corpus_chunks.json`, embeds all 2,467 chunks, inserts into `chunks` table with pgvector embedding, saves FAISS index to `data/processed/chunks/faiss.index`.

Key logic:

```python
import json, faiss, numpy as np
from tqdm import tqdm
from db.models import Chunk, Textbook
from src.utils.embedding_client import EmbeddingClient
from db import get_session

def ingest(source_json: str, textbook_name: str):
    chunks = json.load(open(source_json))
    ec = EmbeddingClient()
    texts = [c["text"] for c in chunks]
    embeddings = ec.embed(texts)          # shape: (2467, 384)

    with get_session() as session:
        tb = Textbook(name=textbook_name, version="2.0", source_path=source_json)
        session.add(tb); session.flush()

        for i, (chunk, emb) in enumerate(tqdm(zip(chunks, embeddings))):
            session.add(Chunk(
                chunk_uid=chunk["chunk_id"],
                textbook_id=tb.id,
                chapter=chunk["chapter"],
                section=chunk["heading"],
                content=chunk["text"],
                embedding=emb.tolist(),
                token_count=len(chunk["text"].split()),
            ))
        session.commit()

    # Save FAISS index
    index = faiss.IndexFlatIP(384)       # inner product on normalized vecs = cosine
    index.add(embeddings.astype(np.float32))
    faiss.write_index(index, "data/processed/chunks/faiss.index")
    print(f"Ingested {len(chunks)} chunks.")
```

### Task 2.1 — Student Agent System Prompt Builder

```python
from config.constants import FSLSM_DIM_LABELS

def build_student_system_prompt(profile: dict) -> str:
    dims = profile["dimensions"]
    instructions = profile["behavioral_instructions"]

    dim_lines = []
    for dim_key, (neg_label, pos_label) in FSLSM_DIM_LABELS.items():
        pole_label = neg_label if dims[dim_key] == -1 else pos_label
        dim_lines.append(f"- **{pole_label}** ({dim_key}): {instructions[dim_key]}")

    return f"""You are a virtual student participating in a learning simulation.

Your Felder-Silverman Learning Style Profile is:
{chr(10).join(dim_lines)}

This profile defines how you prefer to learn, think, and respond. You must express these preferences authentically and consistently in all your responses. When asked questions, always answer from the perspective of a student with this profile — do not break character or acknowledge that you are an AI.

Topic domain: Introductory Machine Learning (neural networks, optimization, gradient descent).
"""
```

### Task 2.4 — ILS Evaluator (Core of Exp1)

**`src/agents/ils_evaluator.py`** — runs the 44-question questionnaire for all 80 agents, across N trials:

```python
import json, re
from pathlib import Path
from tqdm import tqdm
from db.models import Agent, FslsmProfile
from db import get_session
from src.utils.llm_client import LLMClient, LLMProvider
from config.constants import FSLSM_DIMENSIONS, ILS_NUM_QUESTIONS

ILS_QUESTION_DIM_MAP = {
    # 44 questions: 11 per dimension, interleaved
    # act_ref: q1,5,9,13,17,21,25,29,33,37,41
    # sen_int: q2,6,10,14,18,22,26,30,34,38,42
    # vis_ver: q3,7,11,15,19,23,27,31,35,39,43
    # seq_glo: q4,8,12,16,20,24,28,32,36,40,44
}

def run_ils_for_agent(agent: Agent, questions: list, client: LLMClient, trial: int) -> dict:
    """Returns dict: {dimension: score (-11 to 11), raw_responses: [...]}"""
    dim_scores = {d: 0 for d in FSLSM_DIMENSIONS}
    raw = []

    for q in questions:
        response = client.chat(
            system=agent.system_prompt,
            user=build_ils_question_prompt(q),
            max_tokens=10,
        )
        answer = response.content.strip().lower()
        answer = re.search(r'[ab]', answer)
        if answer:
            answer = answer.group()
            pole = q[f"option_{answer}"]["pole"]   # -1 or +1
            dim_scores[q["dimension"]] += pole
        raw.append({"q_num": q["q_num"], "answer": answer, "tokens": response.prompt_tokens})

    # Save raw responses
    out_path = Path(f"results/exp1/raw_responses/{agent.agent_uid}_trial{trial}.json")
    out_path.write_text(json.dumps({"agent_uid": agent.agent_uid, "trial": trial,
                                    "dim_scores": dim_scores, "raw": raw}, indent=2))
    return dim_scores


def run_experiment1(llm_model: str, provider: LLMProvider, num_trials: int = 3):
    questions = json.load(open("data/fslsm/ils_questionnaire.json"))
    client = LLMClient(llm_model, provider, temperature=0.3)

    with get_session() as session:
        agents = session.query(Agent).filter_by(llm_model=llm_model).all()

    results = []
    for agent in tqdm(agents, desc=f"Exp1 [{llm_model}]"):
        for trial in range(1, num_trials + 1):
            dim_scores = run_ils_for_agent(agent, questions, client, trial)
            results.append({
                "agent_uid": agent.agent_uid,
                "trial": trial,
                "assigned": agent.profile.dimensions,
                "detected": {d: (1 if dim_scores[d] >= 0 else -1) for d in FSLSM_DIMENSIONS},
                "raw_scores": dim_scores,
            })

    return results
```

### Task 2.6 — Metrics Computation

**`src/evaluation/metrics.py`** — implement PRA and DAS:

```python
import numpy as np
from config.constants import FSLSM_DIMENSIONS

def profile_recovery_accuracy(results: list[dict]) -> dict:
    """
    PRA per dimension and overall (4D).
    results: list of {assigned: {dim: int}, detected: {dim: int}}
    """
    dim_matches = {d: [] for d in FSLSM_DIMENSIONS}
    for r in results:
        for d in FSLSM_DIMENSIONS:
            dim_matches[d].append(int(r["assigned"][d] == r["detected"][d]))

    pra_per_dim = {d: np.mean(dim_matches[d]) for d in FSLSM_DIMENSIONS}
    pra_4d = np.mean([v for v in pra_per_dim.values()])
    return {"per_dimension": pra_per_dim, "overall_4d": pra_4d}


def dimension_alignment_score(agent_embeddings: np.ndarray,
                               trait_embeddings: np.ndarray) -> float:
    """
    DAS = cosine similarity between agent's aggregated responses
    and Graf (2007) style descriptors for their assigned dimension.
    Both inputs must be L2-normalized.
    """
    return float(np.dot(agent_embeddings, trait_embeddings))
```

```python
import matplotlib.pyplot as plt
import numpy as np

def radar_chart(agent_uid: str, assigned: dict, detected_avg: dict, save_path: str):
    dims = ["act_ref", "sen_int", "vis_ver", "seq_glo"]
    labels = ["Active/Reflective", "Sensing/Intuitive", "Visual/Verbal", "Sequential/Global"]
    assigned_vals = [assigned[d] for d in dims]
    detected_vals = [detected_avg[d] for d in dims]

    angles = np.linspace(0, 2 * np.pi, len(dims), endpoint=False).tolist()
    angles += angles[:1]
    assigned_vals += assigned_vals[:1]
    detected_vals += detected_vals[:1]

    fig, ax = plt.subplots(subplot_kw=dict(polar=True), figsize=(5, 5))
    ax.plot(angles, assigned_vals, 'b-o', label="Assigned Profile", linewidth=2)
    ax.plot(angles, detected_vals, 'r--o', label="ILS Questionnaire Result", linewidth=2)
    ax.set_xticks(angles[:-1])
    ax.set_xticklabels(labels, fontsize=9)
    ax.set_title(f"Agent {agent_uid}", pad=15)
    ax.legend(loc="upper right", bbox_to_anchor=(1.3, 1.1))
    plt.tight_layout()
    plt.savefig(save_path, dpi=150, bbox_inches="tight")
    plt.close()
```

## Prompt Refinement Loop (If PRA < 0.70)

If any model scores below 0.70 PRA on a dimension, iterate on the system prompt:

1. **Increase behavioral specificity** — add concrete example phrases to `behavioral_instructions` (e.g., *"You often say things like: 'Can you show me a worked example?' or 'I want to try this myself.'"*)
2. **Add ILS-aligned example Q&A pairs** in the system prompt (few-shot in-context examples)
3. **Lower temperature** to 0.1 for more deterministic responses
4. **Re-run** the evaluator on a sample of 10 agents before running all 80

Document each iteration with the PRA delta in `results/exp1/metrics/prompt_iterations.md`.
