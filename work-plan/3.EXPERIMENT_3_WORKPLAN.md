> **For use with Claude Code in local IDE**
> Version: 1.0 | Thesis: FSLSM-RAG-MCP Adaptive AI Tutoring System

---

## Overview

This workplan implements the three-condition ablation study (S0 / S1a / S1b) that
answers **RQ3**: Does FSLSM-conditioned MCP tool selection improve efficiency (PTS)
and precision (TSA) over prompt-bloat baseline and unconditioned RAG-MCP?

**Key constraint:** Experiment 3 reuses the exact 4,000 sessions from Experiment 2.
No new LLM calls are needed for session generation — only for the one-time FAISS
tool index build (embedding 15 tool descriptions).

**Tech stack:** Python · PostgreSQL + pgvector · googleapis/mcp-toolbox · FastMCP ·
OpenAI (`text-embedding-3-small` + `gpt-4.1-mini`) · FAISS · SQLite (results log)

```sql
-- db/schema.sql

-- Enable pgvector
CREATE EXTENSION IF NOT EXISTS vector;

-- Tool registry table
-- Stores all 15 MCP tool definitions with their FSLSM dimension tags
CREATE TABLE IF NOT EXISTS mcp_tools (
    tool_id         INTEGER PRIMARY KEY,
    name            TEXT NOT NULL,
    category        TEXT NOT NULL,
    fslsm_dims      TEXT[],          -- e.g. ARRAY['Visual', 'Sequential']
    description     TEXT NOT NULL,
    parameters      JSONB,
    token_cost      INTEGER,
    embedding       vector(1536)     -- populated in Step 2.2
);

-- Session results table
-- One row per (session_id, condition) triple — 3 rows per session
CREATE TABLE IF NOT EXISTS exp3_session_results (
    id                  SERIAL PRIMARY KEY,
    condition           TEXT NOT NULL CHECK (condition IN ('S0','S1a','S1b')),
    session_id          TEXT NOT NULL,
    student_profile     JSONB NOT NULL,
    query               TEXT NOT NULL,
    selected_tool_id    INTEGER REFERENCES mcp_tools(tool_id),
    optimal_tool_id     INTEGER REFERENCES mcp_tools(tool_id),
    tsa_hit             BOOLEAN NOT NULL,
    input_tokens        INTEGER NOT NULL,
    pts_delta           FLOAT NOT NULL,
    created_at          TIMESTAMPTZ DEFAULT NOW()
);

CREATE INDEX IF NOT EXISTS idx_results_condition
    ON exp3_session_results(condition);
CREATE INDEX IF NOT EXISTS idx_results_session
    ON exp3_session_results(session_id);
```

```python
# scripts/01_setup_postgres.py
"""
Creates PostgreSQL schema and loads the 15-tool registry.
Run once before the experiment.
"""
import sys, os
sys.path.insert(0, os.path.dirname(os.path.dirname(__file__)))

import psycopg2
from config import POSTGRES_URL
from exp3_mcp_tools.tool_registry import TOOL_REGISTRY

def setup():
    conn = psycopg2.connect(POSTGRES_URL)
    cur  = conn.cursor()

    # 1. Apply schema
    with open("db/schema.sql") as f:
        cur.execute(f.read())

    # 2. Load tool registry (embeddings added in Step 2.2)
    for t in TOOL_REGISTRY:
        cur.execute("""
            INSERT INTO mcp_tools (tool_id, name, category, fslsm_dims,
                                   description, parameters, token_cost)
            VALUES (%s, %s, %s, %s, %s, %s, %s)
            ON CONFLICT (tool_id) DO UPDATE
                SET description = EXCLUDED.description,
                    token_cost   = EXCLUDED.token_cost
        """, (t.tool_id, t.name, t.category, t.fslsm_dims,
              t.description, str(t.parameters), t.token_cost))

    conn.commit()
    conn.close()
    print(f"[setup_postgres] Schema created. {len(TOOL_REGISTRY)} tools loaded.")

if __name__ == "__main__":
    setup()
```

```yaml
  search-tools-by-embedding:
    kind: postgres-sql
    source: thesis-db
```

### Step 2.1 — Embed tool descriptions and build FAISS index

**Run:** `scripts/02_build_tool_index.py`

```python
# scripts/02_build_tool_index.py
"""
Embeds all 15 tool descriptions with text-embedding-3-small.
Builds FAISS IndexFlatIP (cosine) and saves to disk.
Also backfills embeddings into PostgreSQL mcp_tools table.

Run once. Takes ~5 seconds and costs ~$0.001 in API credits.
"""
import sys, os
sys.path.insert(0, os.path.dirname(os.path.dirname(__file__)))

import psycopg2
import json
from config import POSTGRES_URL, TOOL_INDEX_PATH, TOOL_META_PATH
from exp3_mcp_tools.tool_index import ToolIndex
from exp3_mcp_tools.tool_registry import TOOL_REGISTRY

def build():
    # 1. Build FAISS index and save to disk
    idx = ToolIndex()
    idx.build()
    idx.save(TOOL_INDEX_PATH, TOOL_META_PATH)

    # 2. Backfill embeddings into PostgreSQL (for pgvector search via toolbox)
    from openai import OpenAI
    import numpy as np
    client = OpenAI(api_key=os.environ["OPENAI_API_KEY"])

    conn = psycopg2.connect(POSTGRES_URL)
    cur  = conn.cursor()

    descriptions = [t.description for t in TOOL_REGISTRY]
    resp = client.embeddings.create(model="text-embedding-3-small", input=descriptions)
    vecs = [r.embedding for r in resp.data]

    for tool, vec in zip(TOOL_REGISTRY, vecs):
        cur.execute(
            "UPDATE mcp_tools SET embedding = %s::vector WHERE tool_id = %s",
            (json.dumps(vec), tool.tool_id)
        )

    conn.commit()
    conn.close()
    print(f"[build_tool_index] FAISS index saved. PostgreSQL embeddings updated.")

if __name__ == "__main__":
    build()
```

```python
# scripts/04_dry_run.py
"""
Dry run: processes 100 sessions from Exp2 data through S0 / S1a / S1b.
Prints per-condition TSA and PTS. Applies go/no-go gate.

Session data format (one JSON object per line in sessions.jsonl):
{
  "session_id": "...",
  "student_profile": {"act":1,"ref":0,"sns":0,"int":0,"vis":1,"vrb":0,"seq":0,"glo":0},
  "query": "Explain backpropagation",
  "turn": 1
}
"""
import sys, os
sys.path.insert(0, os.path.dirname(os.path.dirname(__file__)))

import json
from tqdm import tqdm
from config import (EXP2_SESSIONS_PATH, TOOL_INDEX_PATH, TOOL_META_PATH,
                    RESULTS_DB_PATH, DRY_RUN_N,
                    MIN_TSA_S1B_OVER_S0, MIN_TSA_S1B_OVER_S1A, MIN_PTS_S1B)
from exp3_mcp_tools.tool_index import ToolIndex
from exp3_mcp_tools.ablation_runner import AblationRunner, ExperimentLogger

def run_dry():
    # Load index
    idx = ToolIndex()
    idx.load(TOOL_INDEX_PATH, TOOL_META_PATH)
    runner = AblationRunner(tool_index=idx)
    logger = ExperimentLogger(RESULTS_DB_PATH)

    # Load first N sessions from Exp2
    sessions = []
    with open(EXP2_SESSIONS_PATH) as f:
        for line in f:
            sessions.append(json.loads(line.strip()))
            if len(sessions) >= DRY_RUN_N:
                break

    print(f"[dry_run] Running {len(sessions)} sessions across S0 / S1a / S1b...\n")

    for sess in tqdm(sessions, desc="Dry run"):
        results = runner.run_session(
            profile    = sess["student_profile"],
            query      = sess["query"],
            session_id = sess["session_id"],
        )
        for result in results.values():
            logger.log(result)

    # Report
    tsa = logger.tsa_by_condition()
    pts = logger.pts_by_condition()

    print("\n── Dry Run Results ──")
    print(f"  TSA  S0  : {tsa.get('S0',0):.3f}  ({tsa.get('S0',0)*100:.1f}%)")
    print(f"  TSA  S1a : {tsa.get('S1a',0):.3f}  ({tsa.get('S1a',0)*100:.1f}%)")
    print(f"  TSA  S1b : {tsa.get('S1b',0):.3f}  ({tsa.get('S1b',0)*100:.1f}%)")
    print(f"  PTS  S1a : {pts.get('S1a',0):.1f}%")
    print(f"  PTS  S1b : {pts.get('S1b',0):.1f}%")

    # Go/No-Go gate
    s1b_over_s0  = tsa.get("S1b",0) - tsa.get("S0",0)
    s1b_over_s1a = tsa.get("S1b",0) - tsa.get("S1a",0)
    pts_s1b      = pts.get("S1b",0)

    print("\n── Go / No-Go Gate ──")
    go = True
    def gate(label, value, threshold, fmt=".3f"):
        nonlocal go
        ok = value >= threshold
        if not ok: go = False
        symbol = "GO ✓" if ok else "NO-GO ✗"
        print(f"  [{symbol}] {label}: {value:{fmt}} (threshold ≥ {threshold:{fmt}})")

    gate("S1b TSA − S0 TSA  ", s1b_over_s0,  MIN_TSA_S1B_OVER_S0)
    gate("S1b TSA − S1a TSA ", s1b_over_s1a, MIN_TSA_S1B_OVER_S1A)
    gate("PTS S1b (%)",         pts_s1b,      MIN_PTS_S1B, ".1f")

    print()
    if go:
        print("✅  ALL GATES PASSED — proceed to Phase 4 (full run).")
    else:
        print("❌  GATE FAILURE — do NOT proceed to full run.")
        print("    Diagnosis checklist:")
        print("    1. If S1b TSA < S0+15pp → expand GROUND_TRUTH_MAP intent keywords")
        print("    2. If S1b ≈ S1a         → review dim-directive vocabulary in fslsm_query_augmentor.py")
        print("    3. If PTS < 80%         → verify S0 is injecting all 15 schemas")

    logger.close()

if __name__ == "__main__":
    run_dry()
```

## Phase 5 — Metrics & Statistical Analysis

### Step 5.1 — Compute all metrics

**Run:** `scripts/06_compute_metrics.py`

```python
# scripts/06_compute_metrics.py
"""
Computes and prints all Experiment 3 metrics for thesis Table 3.4:
  - TSA per condition (S0, S1a, S1b)
  - PTS per condition
  - Per-dimension TSA breakdown (Visual, Sequential, Active, etc.)
  - S1b vs S0 delta with 95% CI
  - S1b vs S1a delta with 95% CI
  - Cohen's h effect size for TSA comparisons

Saves results/exp3_metrics.json for report generation.
"""
import sys, os
sys.path.insert(0, os.path.dirname(os.path.dirname(__file__)))

import sqlite3, json
import numpy as np
from scipy import stats
from config import RESULTS_DB_PATH

def cohens_h(p1, p2):
    """Cohen's h for two proportions."""
    return 2 * np.arcsin(np.sqrt(p1)) - 2 * np.arcsin(np.sqrt(p2))

def compute():
    conn = sqlite3.connect(RESULTS_DB_PATH)

    def query(sql, params=()):
        return conn.execute(sql, params).fetchall()

    metrics = {}

    # Overall TSA and PTS
    for cond in ["S0","S1a","S1b"]:
        rows = query(
            "SELECT tsa_hit, input_tokens, pts_delta FROM exp3_session_results WHERE condition=?",
            (cond,)
        )
        tsa_hits = [r[0] for r in rows]
        pts_vals = [r[2] for r in rows]
        metrics[cond] = {
            "n": len(tsa_hits),
            "tsa_mean": float(np.mean(tsa_hits)),
            "tsa_se":   float(np.std(tsa_hits) / np.sqrt(len(tsa_hits))),
            "pts_mean": float(np.mean(pts_vals)),
            "pts_se":   float(np.std(pts_vals) / np.sqrt(len(pts_vals))),
        }

    # Per-dimension TSA for S1b
    dim_tsa = {}
    fslsm_dims = ["Active","Reflective","Sensing","Intuitive","Visual","Verbal","Sequential","Global"]
    for dim in fslsm_dims:
        rows = query("""
            SELECT sr.tsa_hit
            FROM exp3_session_results sr
            JOIN mcp_tools mt ON sr.optimal_tool_id = mt.tool_id
            WHERE sr.condition = 'S1b'
              AND mt.fslsm_dims @> ARRAY[?]
        """, (dim,))
        # Note: if using SQLite (not PostgreSQL), filter via Python instead
        if rows:
            hits = [r[0] for r in rows]
            dim_tsa[dim] = float(np.mean(hits))

    metrics["dim_tsa_S1b"] = dim_tsa

    # Statistical comparison S1b vs S0
    s0_hits  = [r[0] for r in query("SELECT tsa_hit FROM exp3_session_results WHERE condition='S0'")]
    s1b_hits = [r[0] for r in query("SELECT tsa_hit FROM exp3_session_results WHERE condition='S1b'")]
    s1a_hits = [r[0] for r in query("SELECT tsa_hit FROM exp3_session_results WHERE condition='S1a'")]

    z_s1b_s0,  p_s1b_s0  = stats.ttest_ind(s1b_hits, s0_hits)
    z_s1b_s1a, p_s1b_s1a = stats.ttest_ind(s1b_hits, s1a_hits)

    metrics["stats"] = {
        "S1b_vs_S0":  {"t": float(z_s1b_s0),  "p": float(p_s1b_s0),
                        "cohens_h": cohens_h(np.mean(s1b_hits), np.mean(s0_hits))},
        "S1b_vs_S1a": {"t": float(z_s1b_s1a), "p": float(p_s1b_s1a),
                        "cohens_h": cohens_h(np.mean(s1b_hits), np.mean(s1a_hits))},
    }

    conn.close()

    # Print thesis table
    print("\n── Experiment 3 Results (Table 3.4) ──\n")
    print(f"{'Condition':<10} {'TSA Mean':>10} {'TSA SE':>8} {'PTS Mean':>10} {'N':>6}")
    print("─" * 50)
    for cond in ["S0","S1a","S1b"]:
        m = metrics[cond]
        print(f"{cond:<10} {m['tsa_mean']*100:>9.1f}% {m['tsa_se']*100:>7.1f}% "
              f"{m['pts_mean']:>9.1f}% {m['n']:>6}")

    print(f"\n── Statistical Tests ──")
    for key, val in metrics["stats"].items():
        sig = "***" if val["p"] < 0.001 else "**" if val["p"] < 0.01 else "*" if val["p"] < 0.05 else "ns"
        print(f"  {key:<15} t={val['t']:+.3f}  p={val['p']:.4f} {sig}  Cohen's h={val['cohens_h']:.3f}")

    print(f"\n── Per-Dimension TSA (S1b) ──")
    for dim, tsa in sorted(dim_tsa.items(), key=lambda x: -x[1]):
        print(f"  {dim:<12} {tsa*100:.1f}%")

    # Save
    os.makedirs("results", exist_ok=True)
    with open("results/exp3_metrics.json","w") as f:
        json.dump(metrics, f, indent=2)
    print("\nSaved results/exp3_metrics.json")

if __name__ == "__main__":
    compute()
```

## Troubleshooting

| Symptom | Likely cause | Fix |
|---|---|---|
| `KeyError: OPENAI_API_KEY` | `.env` not loaded | Add `load_dotenv()` or `export OPENAI_API_KEY=sk-...` |
| `ModuleNotFoundError: faiss` | Not installed | `pip install faiss-cpu` |
| FAISS index returns wrong tool | Tool descriptions too similar | Add stronger distinguishing vocabulary to `description` fields in `tool_registry.py` |
| TSA S1b ≈ TSA S1a | Augmentation not changing retrieval | Print augmented queries; check `DIM_DIRECTIVES` vocabulary overlaps with tool descriptions |
| PostgreSQL connection refused | DB not running | `pg_ctl start` or check Docker container |
| `toolbox` binary not found | Path wrong | Update `TOOLBOX_BINARY` in `config.py` |
| Dry run gate fails on PTS | S0 token count wrong | Verify `s0_prompt_tokens()` returns sum of all 15 tool `token_cost` values |
